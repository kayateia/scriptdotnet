Please consider tutorial lesson getting starting. 
  
using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Text;
using System.Windows.Forms;
using ScriptNET;

namespace Test
{
 public partial class Program
 {

  
   public static in Main()
   {
    List vals = new List();
    vals.AddRange(new int[] { 1, 2, 3, 4 });

    Script script = Script.Compile(@"
            rez = 0;
            foreach (number in numbers)
            rez += number;"
           ); 

    //Adding variable to script's scope
    script.Context.SetItem("numbers", ContextItem.Variable, vals); 

    object rez = script.Execute();
    Console.WriteLine(rez); 
   }
}
